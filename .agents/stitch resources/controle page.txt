        function toggleLED(btn, color) {
            // Remove active classes from all buttons in the remote section
            const buttons = document.querySelectorAll('#remote-mode button');
            buttons.forEach(b => {
                b.classList.remove('active', 'active-glow-orange', 'active-glow-green', 'active-glow-red', 'active-glow-blue', 'active-glow-yellow', 'active-glow-white', 'border-primary-container');
            });
            
            // Add active state to clicked button
            btn.classList.add('active', `active-glow-${color}`, 'border-primary-container');
        }

        function selectChip(btn) {
            const chips = btn.parentElement.querySelectorAll('button');
            chips.forEach(c => {
                c.classList.remove('bg-primary', 'text-on-primary');
                c.classList.add('bg-surface-container', 'text-on-surface-variant');
            });
            btn.classList.remove('bg-surface-container', 'text-on-surface-variant');
            btn.classList.add('bg-primary', 'text-on-primary');
        }

        // Initialize active state for one button
        document.addEventListener('DOMContentLoaded', () => {
            const morningBtn = document.querySelector('button[onclick*="yellow"]');
            if(morningBtn) toggleLED(morningBtn, 'yellow');
        });

        // Simple toggle logic for Dynamic Mode
        const dynamicToggle = document.getElementById('toggle-dynamic');
        dynamicToggle.addEventListener('click', () => {
            const ball = dynamicToggle.querySelector('div');
            if (ball.classList.contains('translate-x-6')) {
                ball.classList.remove('translate-x-6');
                dynamicToggle.classList.replace('bg-white/30', 'bg-slate-300');
            } else {
                ball.classList.add('translate-x-6');
                dynamicToggle.classList.replace('bg-slate-300', 'bg-white/30');
            }
        });
    </script>
